// --- הנה הקוד המתוקן והחסין לקריסות --- async function runScan() { const scanBtn = document.getElementById('scan-btn'); const list = document.getElementById('stock-list'); try { scanBtn.classList.add('scanning'); list.innerHTML = '
SCANNING...
'; // כאן אתה עושה את ה-fetch שלך const response = await fetch('YOUR_PROXY_URL_HERE'); // וודא שה-URL שלך פה תקין const data = await response.json(); // הגנה: אם הנתונים לא מערך, אנחנו לא קורסים if (!Array.isArray(data)) throw new Error('Invalid Data'); // שימוש בתיקון שדיברנו עליו - סינון מניות דפוקות const liveStocks = data.map(q => { if (!q || !q.symbol) return null; return { ticker: q.symbol, price: q.regularMarketPrice || 0, chg: q.regularMarketChangePercent || 0, vol: (q.regularMarketVolume || 0) / 1e6, // ... שאר השדות }; }).filter(s => s !== null); // ציור התוצאות renderStocks(liveStocks); } catch (err) { console.error("Scan failed:", err); list.innerHTML = '
שגיאה בטעינת נתונים.
בדוק את ה-Proxy שלך.
'; } finally { scanBtn.classList.remove('scanning'); } }
